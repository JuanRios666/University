81void read_registers(uint8_t reg, uint8_t *buf, uint16_t len)
82{
83 // For this particular device, we send the device the register we want to read
84 // first, then subsequently read from the device. The register is auto incrementing
85 // so we don't need to keep sending the register we want, just the first.
86
87 reg |= READ_BIT;
88 cs_select();
89 spi_write_blocking(SPI_PORT, &reg, 1);
90 sleep_ms(10);
91 spi_read_blocking(SPI_PORT, 0, buf, len);
92 cs_deselect();
93 sleep_ms(10);

196void calculate_angles(int16_t eulerAngles[2], int16_t accel[3], int16_t gyro[3], uint64_t usSinceLastReading) //Calculates angles based on the accelerometer and gyroscope. Requires usSinceLastReading to use the gyro.
197{
198 long hertz = 1000000/usSinceLastReading;
199
200 if (hertz < 200)
201 {
202 calculate_angles_from_accel(eulerAngles, accel);
203 return;
204 }
205
206 long temp = 1.l/(hertz * 65.5l);
207
208 eulerAngles[0] += gyro[0] * temp;
209 eulerAngles[1] += gyro[1] * temp;
210
211 eulerAngles[0] += eulerAngles[1] * sin(gyro[2] * temp * 0.1f);
212 eulerAngles[1] -= eulerAngles[0] * sin(gyro[2] * temp * 0.1f);
213
214 int16_t accelEuler[2];
215 calculate_angles_from_accel(accelEuler, accel);
216
217 eulerAngles[0] = eulerAngles[0] * 0.9996 + accelEuler[0] * 0.0004;
218 eulerAngles[1] = eulerAngles[1] * 0.9996 + accelEuler[1] * 0.0004;
219}

229void calculate_angles_from_accel(int16_t eulerAngles[2], int16_t accel[3]) //Uses just the direction gravity is pulling to calculate angles.
230{
231 float accTotalVector = sqrt((accel[0] * accel[0]) + (accel[1] * accel[1]) + (accel[2] * accel[2]));
232
233 float anglePitchAcc = asin(accel[1] / accTotalVector) * 57.296;
234 float angleRollAcc = asin(accel[0] / accTotalVector) * -57.296;
235
236 eulerAngles[0] = anglePitchAcc;
237 eulerAngles[1] = angleRollAcc;
238}

249void convert_to_full(int16_t eulerAngles[2], int16_t accel[3], int16_t fullAngles[2]) //Converts from -90/90 to 360 using the direction gravity is pulling
250{
251 if (accel[1] > 0 && accel[2] > 0) fullAngles[0] = eulerAngles[0];
252 if (accel[1] > 0 && accel[2] < 0) fullAngles[0] = 180 - eulerAngles[0];
253 if (accel[1] < 0 && accel[2] < 0) fullAngles[0] = 180 - eulerAngles[0];
254 if (accel[1] < 0 && accel[2] > 0) fullAngles[0] = 360 + eulerAngles[0];
255

257 if (accel[0] < 0 && accel[2] < 0) fullAngles[1] = 180 - eulerAngles[1];
258 if (accel[0] > 0 && accel[2] < 0) fullAngles[1] = 180 - eulerAngles[1];
259 if (accel[0] > 0 && accel[2] > 0) fullAngles[1] = 360 + eulerAngles[1];
